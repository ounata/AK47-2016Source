                        Javascript

                            angular.module('app.component')
                            .controller('MCSValidationController', ['$scope', 'mcsValidationService', function ($scope, mcsValidationService) {
                                var vm = this;                                
                                mcsValidationService.init($scope);
                                vm.dictionaries = {
                                    'c_codE_ABBR_CUSTOMER_GRADE': [
                                        { key: 'G1', value: '一年级' },
                                        { key: 'G2', value: '二年级' },
                                        { key: 'G3', value: '三年级' },
                                        { key: 'G4', value: '四年级' },
                                        { key: 'G5', value: '五年级' },
                                        { key: 'G6', value: '六年级' }
                                    ]
                                };
                                mcs.util.merge(vm.dictionaries);
                                vm.save = function () {
                                    if (mcsValidationService.run($scope)) {
                                        alert("验证成功！");
                                        //....
                                    }
                                };
                            }])

参数说明

规则 默认值 示例
required 设置输入项为必填(选) <mcs-input model="vm.demo.required" css="col-md-12" required/>
required-level 设置级联下拉框为必选到指定级 <mcs-cascading-select caption="省份,城市,区县" data="vm.items" model="vm.model" path="true" async="false" required-level="2"/>
minlength 设置输入项的最小长度 <mcs-input model="vm.demo.minlength" css="col-md-12" minlength="3"/>
maxlength 设置输入项的最大长度 <mcs-input model="vm.demo.maxlength" css="col-md-12" maxlength="20"/>
min 设置输入项的最小值（可为小数） <mcs-input model="vm.demo.min" css="col-md-12" min="4"/>
max 设置输入项的最大值（可为小数） <mcs-input model="vm.demo.max" css="col-md-12" max="100"/>
less 设置输入项的最小值（不包含本身，可为小数） <mcs-input model="vm.demo.less" css="col-md-12" less="4"/>
great 设置输入项的最大值（不包含本身，可为小数） <mcs-input model="vm.demo.great" css="col-md-12" great="100"/>
between 设置输入项的输入范围 <mcs-input model="vm.demo.between" css="col-md-12" between="4,100"/>
positive 设置输入项必须为正整数（含0） <mcs-input model="vm.demo.positive" css="col-md-12" positive/>
negative 设置输入项必须为负整数（含0） <mcs-input model="vm.demo.negative" css="col-md-12" negative/>
number 设置输入项为小数 <mcs-input model="vm.demo.number" css="col-md-12" number/>
currency 设置输入项为货币值（只保留两位小数） <mcs-input model="vm.demo.currency" css="col-md-12" currency/>
phone 设置输入项必须为电话（含手机或座机） <mcs-input model="vm.demo.phone" css="col-md-12" phone/>
email 设置输入项为电子邮件 <mcs-input model="vm.demo.email" css="col-md-12" email/>
idcard 设置输入项为身份证号 <mcs-input model="vm.demo.idcard" css="col-md-12" idcard/>
data-validation-[rule]-message 设置输入项为必填并带自定义的message,这里的[rule]可换成对应的规则名称，如：required, minlength <mcs-input model="vm.demo.required" css="col-md-12" required data-validation-required-message="请输入数据，否则不能保存！"/>
validate 设置自定义验证 <mcs-input model="vm.demo.validate" css="col-md-12" validate="vm.checkName()"/>
