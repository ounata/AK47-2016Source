与树控件配合的使用说明:

在通过树控件选取节点后拿到已经选取的节点数据，加入到tags-input 绑定的数据集中，例如：vm.teachers = [{ "teacherId": "1", "name": "tom" }];

示例代码

HTML

 <tags-input class="mcs-tags-input-lg" ng-disable="true" ng-model="vm.teachers" key-property="teacherId" display-property="name" placeholder="姓名">
    <auto-complete min-length="1" source="vm.queryTeacherList($query)"></auto-complete>
  </tags-input>

JS

(function () {
    'use strict';
    angular.module('app.component').controller('MCSAutoCompleteController', [
        '$scope', '$http',

        function ($scope, $http) {
            var vm = this;
            $scope.vm = vm;

            vm.tags = [{
                teacherId: '1',
                name: 'tom'
            }, {
                teacherId: '2',
                name: 'jack'
            }, {
                teacherId: '3',
                name: 'jerry'
            }];

            vm.queryTeacherList = function (query) {
                var result = [];

                vm.tags.forEach(function (item) {
                    if (item.name.indexOf(query) > -1) {
                        result.push(item);
                    }

                });

                return result;
            };

        }
    ]);
})();
