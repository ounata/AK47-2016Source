应用范例

基于路由的页面动态加载

ng-incude方式加载页面

示例代码

JS
angular.module('app', ['ui.bootstrap', 'ui.router'])
.config(['$stateProvider', '$urlRouterProvider', function($stateProvider, $urlRouterProvider) {
$stateProvider
    .state('home', {
        url: "/home",
        templateUrl: "view/home.html"
    })
    .state('one', {
        url: "/one",
        templateUrl: "view/home.one.html",
        controller: function($scope) {
            $scope.items = ["A", "List", "Of", "Items"];
        }
    })
    .state('two', {
        url: "/two",
        templateUrl: "view/home.two.html",
        controller: function($scope) {
            $scope.items = ["A", "List", "Of", "Items"];
        }
    });
$urlRouterProvider.otherwise('/home');
}])
.controller('appCtrl', ['$rootScope', '$scope', 'dialogs', '$timeout', function($rootScope, $scope, dialogs, $timeout) {
var vm = {};
$scope.vm = vm;
vm.tabs = [{
    title: 'one',
    active: false
}, {
    title: 'two',
    active: true
}]
vm.changeTab = function(tab) {
    tab.active = tab.active ? false : true;
    vm.tabs[0].active = true;
}
}])

参数说明

参数 用途 取值范围 示例
heading 设定每个tab项的标题 string heading="{{tab.title}}"
ui-sref 设定每个tab项点击时动态加载页面视图的路由 string ui-sref="{{tab.route}}"
active 设定被打开的tab项 string active="tab.active"
